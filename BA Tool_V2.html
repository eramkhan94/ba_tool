                    <p class="text-gray-500">Data architecture models, data object catalogs, and data flow diagrams would appear here.</p>

                </div>

            </div>


        </main>

    </div>


    <script>

        document.addEventListener('DOMContentLoaded', function() {

            let processesChart;

            const container = document.getElementById('flowchart-container');


            const processHierarchy = [

                { 

                    id: "1", name: "Client Strategy Development", isMain: true, performedBy: "Sales",

                    subProcesses: [

                        { id: "1.1", name: "Identify Target Market", description: "Identify target market for offering", performedBy: "Sales", system: "N/A", dataIn: "Client Business", dataOut: "Client Business ID" },

                        { id: "1.2", name: "Developing Proposition", description: "Develop proposition based on target market", performedBy: "Sales", system: "N/A", dataIn: "Client Business", dataOut: "Client Business" },

                        { id: "1.3", name: "Identifying & Planning Segments", description: "Identify segments within TG and plan sales strategy", performedBy: "Sales", system: "CRM", dataIn: "Client Business", dataOut: "Client Business" }

                    ]

                },

                { 

                    id: "2", name: "Investment Guidelines", isMain: true, performedBy: "Fund Manager",

                    subProcesses: [

                        { id: "2.1", name: "Investment Restrictions", description: "Define guidelines based on client preferences", performedBy: "Fund Manager", system: "N/A", dataIn: "N/A", dataOut: "Client Business" },

                        { id: "2.2", name: "Compliance Restrictions", description: "Define guidelines based on compliance preferences", performedBy: "Risk Team", system: "N/A", dataIn: "N/A", dataOut: "N/A" }

                    ]

                }

            ];


            const expansionState = { '1': false, '2': false };

            

            // --- View Switching Logic ---

            const views = document.querySelectorAll('.view');

            const links = document.querySelectorAll('.sidebar-link');

            

            function showView(viewId) {

                views.forEach(view => view.classList.remove('active'));

                links.forEach(link => link.classList.remove('active'));


                const activeView = document.getElementById(viewId);

                if (activeView) activeView.classList.add('active');


                const activeLink = document.querySelector(`.sidebar-link[data-view="${viewId}"]`);

                if (activeLink) activeLink.classList.add('active');

                

                if (viewId === 'dashboard') {

                    drawProcessesPerCapabilityChart();

                }

                if (viewId === 'processes') {

                    drawFlowchart();

                }

            }


            links.forEach(link => {

                link.addEventListener('click', (e) => {

                    e.preventDefault();

                    const viewId = e.currentTarget.dataset.view;

                    showView(viewId);

                });

            });


            // --- Chart.js for Dashboard ---

            function drawProcessesPerCapabilityChart() {

                const ctx = document.getElementById('processesPerCapabilityChart');

                if (!ctx) return;

                if (processesChart) processesChart.destroy();


                processesChart = new Chart(ctx.getContext('2d'), {

                    type: 'bar',

                    data: {

                        labels: ['Investment Mgmt', 'Risk Mgmt', 'Client Reporting', 'Compliance', 'Trade Execution', 'Corporate Support'],

                        datasets: [{

                            label: '# of Processes',

                            data: [45, 25, 22, 38, 32, 18],

                            backgroundColor: 'rgba(59, 130, 246, 0.6)',

                            borderColor: 'rgba(59, 130, 246, 1)',

                            borderWidth: 1

                        }]

                    },

                    options: {

                        responsive: true,

                        maintainAspectRatio: false,

                        scales: { y: { beginAtZero: true, title: { display: true, text: 'Number of Processes' } } },

                        plugins: { legend: { display: false } }

                    }

                });

            }


            // --- Flowchart Drawing Logic ---

            function drawFlowchart() {

                if (!container) return;

                container.innerHTML = ''; // Clear previous content


                const visibleNodes = [];

                processHierarchy.forEach(mainProcess => {

                    if (expansionState[mainProcess.id]) {

                        visibleNodes.push(...mainProcess.subProcesses);

                    } else {

                        visibleNodes.push(mainProcess);

                    }

                });


                const svgNs = "http://www.w3.org/2000/svg";

                const svg = document.createElementNS(svgNs, 'svg');

                svg.setAttribute('class', 'absolute top-0 left-0 w-full h-full');

                svg.setAttribute('style', 'z-index: 0;');

                

                const defs = document.createElementNS(svgNs, 'defs');

                const marker = document.createElementNS(svgNs, 'marker');

                marker.setAttribute('id', 'arrowhead');

                marker.setAttribute('viewBox', '0 0 10 10');

                marker.setAttribute('refX', '8');

                marker.setAttribute('refY', '5');

                marker.setAttribute('markerWidth', '6');

                marker.setAttribute('markerHeight', '6');

                marker.setAttribute('orient', 'auto-start-reverse');

                const polygon = document.createElementNS(svgNs, 'polygon');

                polygon.setAttribute('points', '0 0, 10 5, 0 10');

                polygon.setAttribute('fill', '#6b7280');

                marker.appendChild(polygon);

                defs.appendChild(marker);

                svg.appendChild(defs);

                container.appendChild(svg);


                const nodeWidth = 180;

                const horizontalGap = 80;

                const startX = 50;

                const startY = 150;


                visibleNodes.forEach((process, index) => {

                    const nodeEl = document.createElement('div');

                    nodeEl.id = `process-node-${process.id.replace('.', '-')}`;

                    nodeEl.dataset.id = process.id;

                    

                    let roleClass = '';

                    if (process.performedBy === 'Sales') roleClass = 'node-sales';

                    else if (process.performedBy === 'Fund Manager') roleClass = 'node-fund-manager';

                    else if (process.performedBy === 'Risk Team') roleClass = 'node-risk-team';


                    nodeEl.className = `flowchart-node ${roleClass} ${process.isMain ? 'is-main' : ''}`;

                    nodeEl.style.left = `${startX + index * (nodeWidth + horizontalGap)}px`;

                    nodeEl.style.top = `${startY}px`;


                    nodeEl.innerHTML = `

                        <span class="font-semibold text-gray-800">${process.id} ${process.name}</span>

                        <div class="flowchart-tooltip">

                            <div class="grid grid-cols-2 gap-x-4 gap-y-2 text-sm">

                                <strong class="text-gray-400">Description:</strong> <span class="text-white">${process.description || 'N/A'}</span>

                                <strong class="text-gray-400">Performed By:</strong> <span class="text-white">${process.performedBy || 'N/A'}</span>

                                <strong class="text-gray-400">System:</strong> <span class="text-white">${process.system || 'N/A'}</span>

                                <strong class="text-gray-400">Data In:</strong> <span class="text-white">${process.dataIn || 'N/A'}</span>

                                <strong class="text-gray-400">Data Out:</strong> <span class="text-white">${process.dataOut || 'N/A'}</span>

                            </div>

                        </div>

                    `;

                    container.appendChild(nodeEl);

                });


                // Draw arrows

                setTimeout(() => {

                    for (let i = 0; i < visibleNodes.length - 1; i++) {

                        const fromNode = visibleNodes[i];

                        const toNode = visibleNodes[i+1];

                        const fromEl = document.getElementById(`process-node-${fromNode.id.replace('.', '-')}`);

                        const toEl = document.getElementById(`process-node-${toNode.id.replace('.', '-')}`);

                        

                        if (fromEl && toEl) {

                            const fromRect = fromEl.getBoundingClientRect();

                            const toRect = toEl.getBoundingClientRect();

                            const containerRect = container.getBoundingClientRect();


                            const x1 = fromRect.right - containerRect.left;

                            const y1 = fromRect.top - containerRect.top + fromRect.height / 2;

                            const x2 = toRect.left - containerRect.left;

                            const y2 = toRect.top - containerRect.top + toRect.height / 2;


                            const line = document.createElementNS(svgNs, 'line');

                            line.setAttribute('x1', x1);

                            line.setAttribute('y1', y1);

                            line.setAttribute('x2', x2 - 8);

                            line.setAttribute('y2', y2);

                            line.setAttribute('stroke', '#6b7280');

                            line.setAttribute('stroke-width', '2');

                            line.setAttribute('marker-end', 'url(#arrowhead)');

                            svg.appendChild(line);

                        }

                    }

                }, 0);

            }


            if (container) {

                container.addEventListener('click', (e) => {

                    const nodeEl = e.target.closest('.flowchart-node.is-main');

                    if (nodeEl) {

                        const nodeId = nodeEl.dataset.id;

                        if (expansionState.hasOwnProperty(nodeId)) {

                            expansionState[nodeId] = !expansionState[nodeId];

                            drawFlowchart();

                        }

                    }

                });

            }


            // Initial calls

            showView('dashboard');

        });

    </script>

</body>

</html>
